SQL & Database Management

Expertise Level: Advanced (7+ years)

Capabilities:

-- Complex queries
SELECT 
    f.facility_name,
    COUNT(DISTINCT p.patient_id) as total_patients,
    AVG(v.bp_systolic) as avg_systolic,
    SUM(CASE WHEN v.bp_controlled = 'Yes' THEN 1 ELSE 0 END) * 100.0 / COUNT(*) as control_rate
FROM facilities f
JOIN patients p ON f.facility_id = p.facility_id
JOIN visits v ON p.patient_id = v.patient_id
WHERE v.visit_date >= '2024-01-01'
GROUP BY f.facility_name
HAVING COUNT(DISTINCT p.patient_id) > 10
ORDER BY control_rate DESC;

-- Database design
CREATE TABLE patients (
    patient_id SERIAL PRIMARY KEY,
    facility_id INTEGER REFERENCES facilities(facility_id),
    enrollment_date DATE NOT NULL,
    diagnosis VARCHAR(100),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Indexes for performance
CREATE INDEX idx_patients_facility ON patients(facility_id);
CREATE INDEX idx_visits_date ON visits(visit_date);
