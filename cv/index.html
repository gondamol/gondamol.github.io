(function() {
  var buttons = Array.from(document.querySelectorAll('.cv-filter-btn'));
  var items = Array.from(document.querySelectorAll('.cv-exp-item'));
  var count = document.getElementById('cvVisibleCount');
  var search = document.getElementById('cvKeywordSearch');
  var active = 'all';

  function passFilter(item) {
    var tags = (item.dataset.tags || '').split(',');
    if (active === 'all') return true;
    return tags.indexOf(active) !== -1;
  }

  function passSearch(item) {
    var term = search && search.value ? search.value.trim().toLowerCase() : '';
    if (!term) return true;
    var hay = (item.dataset.search || '').toLowerCase();
    return hay.indexOf(term) !== -1;
  }

  function render() {
    var visible = 0;
    items.forEach(function(item) {
      var match = passFilter(item) && passSearch(item);
      item.style.display = match ? '' : 'none';
      if (match) visible += 1;
    });
    if (count) count.textContent = String(visible);
  }

  buttons.forEach(function(btn) {
    btn.addEventListener('click', function() {
      active = btn.dataset.filter || 'all';
      buttons.forEach(function(other) {
        other.classList.remove('active');
        other.setAttribute('aria-selected', 'false');
      });
      btn.classList.add('active');
      btn.setAttribute('aria-selected', 'true');
      render();
    });
  });

  if (search) {
    search.addEventListener('input', render);
  }

  render();
})();
</script>
